g1 <- ds1 %>% 
  group_by(year,sex,racethnicity,age_group) %>% 
  summarise(
    n_people = sum(count, na.rm = TRUE)
  ) %>% 
  ungroup() %>% 
  ggplot(aes(x = year, y = n_people, color = sex, group = interaction(racethnicity, sex))) +
  geom_line() +
  facet_grid(racethnicity ~ age_group, scales = "free_y")
